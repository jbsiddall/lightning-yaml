import speedRaw from '../../../data/benchmarks/speed.yaml?raw'
import ChartCard from '../../../components/charts/ChartCard.astro'
import {
  newestRun,
  parseRuns,
  pickWorkloads,
  speedGroups,
  seriesFor,
  legendFor,
  groupedBarSVG,
  speedTableHtml,
  speedTrend,
  trendLegend,
  lineChartSVG,
  runDates,
  xTicks,
  formatRatio,
  LIBRARY_ORDER,
  presentIn,
} from '../../../lib/benchmarks'

export const speed = newestRun(speedRaw)
export const speedRuns = parseRuns(speedRaw)

export const chartWorkloadNames = [
  'small-records',
  'medium-records',
  'large-records',
  'xlarge-records',
  'yaml-rich-medium',
]
export const parseChartWorkloads = pickWorkloads(speed.operations.parse, chartWorkloadNames)
export const parseSeries = seriesFor(speed.libraries, presentIn(LIBRARY_ORDER, parseChartWorkloads))
export const parseLegend = legendFor(speed.libraries, presentIn(LIBRARY_ORDER, parseChartWorkloads))
export const parseSvg = groupedBarSVG({
  id: 'chart-parse',
  title: 'Parse time by workload (relative to fastest, per row)',
  series: parseSeries,
  groups: speedGroups(parseChartWorkloads),
  unit: 'ns',
  lowerIsBetter: true,
  labelStyle: 'ratio-to-best',
})

export const trendWorkload = 'large-records'
export const trendOrder = presentIn(LIBRARY_ORDER, pickWorkloads(speed.operations.parse, [trendWorkload]))
export const parseTrendSeries = speedTrend(speedRuns, 'parse', trendWorkload, trendOrder)
export const parseTrendLegend = trendLegend(parseTrendSeries)
export const parseTrendSvg = lineChartSVG({
  id: 'trend-parse',
  title: `Parse time vs fastest over time — ${trendWorkload}`,
  series: parseTrendSeries,
  ticks: xTicks(runDates(speedRuns)),
  n: speedRuns.length,
  yFormat: formatRatio,
  higherIsBetter: false,
  yScaleType: 'log',
})

export const summaryTableHtml = speedTableHtml(speed.operations.parse, speed.libraries, presentIn(LIBRARY_ORDER, speed.operations.parse))
export const provenanceLine = `Measured on ${speed.env.cpu} (${speed.env.clk}) · ${speed.env.runtime} · generated ${speed.generated} · source ${speed.source}`

How fast **Lightning YAML** turns text into JavaScript values, against `js-yaml`, `yaml`
(eemeli), and the `JSON` baseline on workloads JSON can read. Every chart is a static image —
inline SVG rendered at build time, no client JavaScript.

<p style="color:var(--sl-color-gray-3);font-family:var(--sl-font-mono);font-size:0.82em">{provenanceLine}</p>

  heading="Parse time by workload"
  note="Lower is better. Bars are proportional to time; each workload is scaled to its own slowest parser (rows span microseconds to seconds, so compare bars within a row, not across rows). Bar labels show time relative to the fastest parser in that row (×) — exact avg ns per parse() call is in the summary table below."
  legend={parseLegend}
  svg={parseSvg}
/>

(anchors and `!!binary`) — the full 13-workload breakdown is in the
[summary table](#summary) below.

## Over time

<ChartCard
  heading={`Parse time vs fastest, run over run — ${trendWorkload}`}
  note="Lower is better, log scale. Absolute nanoseconds drift with CI-runner noise, so this tracks each parser's time as a multiple of the fastest parser in the same run (×) — a machine-invariant figure. The y-axis is logarithmic because the parsers span about two orders of magnitude; the fastest sits at 1× by construction, and a rising Lightning YAML line would mean a regression. One document per published run, oldest to newest."
  legend={parseTrendLegend}
  svg={parseTrendSvg}
/>

Locally (with only the committed sample run) this renders a single point; the deployed site
overlays the full [`benchmark-data`](https://github.com/jbsiddall/lightning-yaml/tree/benchmark-data)
history, so it renders as lines.
