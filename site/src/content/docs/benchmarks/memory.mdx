import memoryRaw from '../../../data/benchmarks/memory.yaml?raw'
import ChartCard from '../../../components/charts/ChartCard.astro'
import {
  newestRun,
  parseRuns,
  pickWorkloads,
  memoryGroups,
  seriesFor,
  legendFor,
  groupedBarSVG,
  memoryTrend,
  trendLegend,
  lineChartSVG,
  runDates,
  xTicks,
  formatMB,
  LIBRARY_ORDER,
  presentIn,
} from '../../../lib/benchmarks'

export const memory = newestRun(memoryRaw)
export const memoryRuns = parseRuns(memoryRaw)

export const memoryWorkloadNames = ['small-records', 'large-records', 'xlarge-records']
export const memoryChartWorkloads = pickWorkloads(memory.operations.parse, memoryWorkloadNames)
export const memorySeries = seriesFor(memory.libraries, presentIn(LIBRARY_ORDER, memoryChartWorkloads))
export const memoryLegend = legendFor(memory.libraries, presentIn(LIBRARY_ORDER, memoryChartWorkloads))
export const memorySvg = groupedBarSVG({
  id: 'chart-memory',
  title: 'Peak memory by workload (parse, resident set size)',
  series: memorySeries,
  groups: memoryGroups(memoryChartWorkloads),
  unit: 'MB',
  lowerIsBetter: true,
})

export const trendWorkload = 'xlarge-records'
export const trendOrder = presentIn(LIBRARY_ORDER, pickWorkloads(memory.operations.parse, [trendWorkload]))
export const memoryTrendSeries = memoryTrend(memoryRuns, trendWorkload, trendOrder)
export const memoryTrendLegend = trendLegend(memoryTrendSeries)
export const memoryTrendSvg = lineChartSVG({
  id: 'trend-memory',
  title: `Peak memory over time — ${trendWorkload}`,
  series: memoryTrendSeries,
  ticks: xTicks(runDates(memoryRuns)),
  n: memoryRuns.length,
  yFormat: formatMB,
  higherIsBetter: false,
})

export const provenanceLine = `Peak RSS across ${memory.iterations} iterations · generated ${memory.generated} · source ${memory.source}`

Whole-process peak resident set (`process.resourceUsage().maxRSS`) while parsing, in isolated
child processes so one library's allocation can't taint another's. Peak RSS — not timing — is
the stable memory figure, so unlike speed these are shown as absolute megabytes.

<p style="color:var(--sl-color-gray-3);font-family:var(--sl-font-mono);font-size:0.82em">{provenanceLine}</p>

<ChartCard
  heading="Peak memory by workload (parse)"
  note="Lower is better. Bars are proportional to peak RSS; each workload is scaled to its own slowest parser, so compare bars within a row, not across rows. Peak resident set size, fixed at 25 iterations."
  legend={memoryLegend}
  svg={memorySvg}
/>

## Over time

<ChartCard
  heading={`Peak memory, run over run — ${trendWorkload}`}
  note="Lower is better. Absolute peak resident set size (MB) on the largest workload, one document per published run, oldest to newest."
  legend={memoryTrendLegend}
  svg={memoryTrendSvg}
/>

Locally (with only the committed sample run) this renders a single point; the deployed site
overlays the full [`benchmark-data`](https://github.com/jbsiddall/lightning-yaml/tree/benchmark-data)
history, so it renders as lines.
