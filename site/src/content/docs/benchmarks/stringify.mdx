import speedRaw from '../../../data/benchmarks/speed.yaml?raw'
import ChartCard from '../../../components/charts/ChartCard.astro'
import {
  newestRun,
  parseRuns,
  pickWorkloads,
  speedGroups,
  seriesFor,
  legendFor,
  groupedBarSVG,
  speedTableHtml,
  speedTrend,
  trendLegend,
  lineChartSVG,
  runDates,
  xTicks,
  formatRatio,
  LIBRARY_ORDER,
  presentIn,
} from '../../../lib/benchmarks'

export const speed = newestRun(speedRaw)
export const speedRuns = parseRuns(speedRaw)

export const chartWorkloadNames = [
  'small-records',
  'medium-records',
  'large-records',
  'xlarge-records',
  'yaml-rich-medium',
]
export const stringifyChartWorkloads = pickWorkloads(speed.operations.stringify, chartWorkloadNames)
export const stringifySeries = seriesFor(speed.libraries, presentIn(LIBRARY_ORDER, stringifyChartWorkloads))
export const stringifyLegend = legendFor(speed.libraries, presentIn(LIBRARY_ORDER, stringifyChartWorkloads))
export const stringifySvg = groupedBarSVG({
  id: 'chart-stringify',
  title: 'Stringify time by workload (relative to fastest, per row)',
  series: stringifySeries,
  groups: speedGroups(stringifyChartWorkloads),
  unit: 'ns',
  lowerIsBetter: true,
  labelStyle: 'ratio-to-best',
})

export const trendWorkload = 'large-records'
export const trendOrder = presentIn(LIBRARY_ORDER, pickWorkloads(speed.operations.stringify, [trendWorkload]))
export const stringifyTrendSeries = speedTrend(speedRuns, 'stringify', trendWorkload, trendOrder)
export const stringifyTrendLegend = trendLegend(stringifyTrendSeries)
export const stringifyTrendSvg = lineChartSVG({
  id: 'trend-stringify',
  title: `Stringify time vs fastest over time — ${trendWorkload}`,
  series: stringifyTrendSeries,
  ticks: xTicks(runDates(speedRuns)),
  n: speedRuns.length,
  yFormat: formatRatio,
  higherIsBetter: false,
  yScaleType: 'log',
})

export const summaryTableHtml = speedTableHtml(speed.operations.stringify, speed.libraries, presentIn(LIBRARY_ORDER, speed.operations.stringify))
export const provenanceLine = `Measured on ${speed.env.cpu} (${speed.env.clk}) · ${speed.env.runtime} · generated ${speed.generated} · source ${speed.source}`

The **js-yaml (tuned)** series is the same library dumping with the lean `CORE` schema
instead of its heavyweight default (YAML 1.1 plus timestamp/binary/omap/set). On
JSON-shaped data that runs ~1.3–1.5× faster and round-trips identically — so it, not the
default, is the fair yardstick for how fast js-yaml *can* stringify this data. It's shown
on the stringify chart only: the rich workloads genuinely need the full schema, where the
"tuned" config would be identical to the default.
