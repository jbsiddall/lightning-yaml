import speedRaw from '../../data/benchmarks/speed.yaml?raw'
import memoryRaw from '../../data/benchmarks/memory.yaml?raw'
import conformanceRaw from '../../data/benchmarks/conformance.yaml?raw'
import ChartCard from '../../components/charts/ChartCard.astro'
import {
  newestRun,
  groupedBarSVG,
  barSVG,
  pickWorkloads,
  speedGroups,
  memoryGroups,
  seriesFor,
  legendFor,
  conformanceItems,
  parseTimeTableHtml,
  conformanceTableHtml,
  LIBRARY_ORDER,
} from '../../lib/benchmarks'

export const speed = newestRun(speedRaw)
export const memory = newestRun(memoryRaw)
export const conformance = newestRun(conformanceRaw)

export const parseWorkloadNames = [
  'small-records',
  'medium-records',
  'large-records',
  'xlarge-records',
  'yaml-rich-medium',
]
export const memoryWorkloadNames = ['small-records', 'large-records', 'xlarge-records']

export const speedSeries = seriesFor(speed.libraries, LIBRARY_ORDER)
export const speedLegend = legendFor(speed.libraries, LIBRARY_ORDER)
export const memoryLegend = legendFor(memory.libraries, LIBRARY_ORDER)

export const parseSvg = groupedBarSVG({
  id: 'chart-parse',
  title: 'Parse time by workload (avg, ns per iteration)',
  series: speedSeries,
  groups: speedGroups(pickWorkloads(speed.operations.parse, parseWorkloadNames)),
  unit: 'ns',
  lowerIsBetter: true,
  logScale: true,
})

export const stringifySvg = groupedBarSVG({
  id: 'chart-stringify',
  title: 'Stringify time by workload (avg, ns per iteration)',
  series: speedSeries,
  groups: speedGroups(pickWorkloads(speed.operations.stringify, parseWorkloadNames)),
  unit: 'ns',
  lowerIsBetter: true,
  logScale: true,
})

export const memorySvg = groupedBarSVG({
  id: 'chart-memory',
  title: 'Peak memory by workload (parse, resident set size)',
  series: seriesFor(memory.libraries, LIBRARY_ORDER),
  groups: memoryGroups(pickWorkloads(memory.operations.parse, memoryWorkloadNames)),
  unit: 'MB',
  lowerIsBetter: true,
  logScale: true,
})

export const conformanceSvg = barSVG({
  id: 'chart-conformance',
  title: 'YAML test-suite conformance (pass rate)',
  items: conformanceItems(conformance.results),
  higherIsBetter: true,
})

export const provenanceLine = `Measured on ${speed.env.cpu} (${speed.env.clk}) · ${speed.env.runtime} · generated ${speed.generated} · source ${speed.source}`

export const summaryTableHtml = parseTimeTableHtml(speed.operations.parse, speed.libraries, LIBRARY_ORDER)
export const conformanceTable = conformanceTableHtml(conformance.results)

Speed, memory, and spec-conformance for **Lightning YAML** against the other parsers this
repo benchmarks itself against: `js-yaml`, `yaml` (eemeli), and the `JSON` baseline where a
workload is JSON-shaped enough for it to compete. Every chart below is a static image —
inline SVG rendered at build time, no client JavaScript.

<p style="color:var(--sl-color-gray-3);font-family:var(--sl-font-mono);font-size:0.82em">{provenanceLine}</p>

These numbers are a hand-captured, representative snapshot — not a live feed. Once the CI
benchmark job is wired up to publish runs to the `benchmark-data` branch, this page will
read that history automatically and the charts will track it without further changes.

## Conformance

<ChartCard
  heading="YAML test-suite conformance"
  note={`Higher is better. Pass rate across ${conformance.suite_total} cases from the official yaml-test-suite.`}
  svg={conformanceSvg}
/>

`lightning-yaml` passes 97.6% of the suite — including all 91 negative (should-fail)
cases — ahead of `yaml` at 97.1% and `js-yaml` at 94.9%. Merge keys (`<<`), absent from the
test corpus, are the only unimplemented construct.
