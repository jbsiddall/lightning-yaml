export const speedRuns = parseRuns(speedRaw, SpeedDocSchema)
export const speed = canonicalRun(speedRuns)
export const speedRuntimes = availableRuntimes(speedRuns)
export const curated = ['small-records', 'medium-records', 'large-records', 'xlarge-records', 'yaml-rich-medium']
export const speedTrendWl = 'large-records'
export const speedSections = speedRuntimes.map((r) =>
  speedFamilySection(r.doc, speedRuns, curated, speedTrendWl, speedRuntimes.length > 1 ? `-${r.family}` : ''),

export const memoryRuns = parseRuns(memoryRaw, MemoryDocSchema)
export const memory = newestOf(memoryRuns)

export const memOrderIds = LIBRARY_ORDER.filter((id) => id !== 'js-yaml-tuned')
export const memTrendWl = 'xlarge-records'

  id: 'bar-parse-memory',
  title: 'Peak memory by workload (parse, resident set size)',

})

  id: 'trend-parse-memory',
  title: `Peak memory over time — ${memTrendWl}`,

  n: memoryRuns.length,
  yFormat: formatMB,
  higherIsBetter: false,
})

  id: 'bar-dump-memory',
  title: 'Peak memory by workload (stringify, resident set size)',
  series: seriesFor(memory.libraries, dumpMemOrder),

export const dumpMemTrendSvg = lineChartSVG({
  id: 'trend-dump-memory',

  id: 'bar-conformance',
  title: 'YAML test-suite conformance (pass rate)',
  items: conformanceItems(conformance.results),

  id: 'trend-conformance',
  title: 'Conformance pass rate over time',
  series: confTrendSeries,
  ticks: xTicks(runDates(conformanceRuns)),
  n: conformanceRuns.length,
  yFormat: formatPercent,
  higherIsBetter: true,
  domainMin: confDomainMin,
  domainMax: 100,

export const bundleSizeBundlerCount = Object.keys(bundleSize.env.bundlers).length
export const bundleSizeChartItems = bundleSizeItems(bundleSize)
export const bundleSizeSvg = barSVG({
  id: 'bar-bundle-size',
  title: 'Bundle size (browser, minified + gzip, smallest across bundlers)',
  items: bundleSizeChartItems,
  higherIsBetter: false,

export const bundleTrendSeries = bundleSizeTrend(bundleSizeRuns, bundleOrder)
export const bundleTrendMax = Math.max(...bundleTrendSeries.flatMap((s) => s.points.map((p) => p.y)), 1)
export const bundleTrendSvg = lineChartSVG({
  id: 'trend-bundle-size',
  title: 'Bundle size over time (smallest gzip across bundlers)',
  series: bundleTrendSeries,
  ticks: xTicks(runDates(bundleSizeRuns)),

})

<p style="color:var(--sl-color-gray-3);font-family:var(--sl-font-mono);font-size:0.82em">{provenanceLine}</p>

How to read these:

- **Bar charts (latest run)** scale each workload row to its own slowest parser, so compare
  bars *within* a row, not across rows. Speed bars are labelled with time relative to the
  fastest parser in that row (`×`); memory bars are labelled with the absolute peak RSS.
- **Trend charts (over time)** render one document per published run, oldest to newest.
  Absolute nanoseconds drift with CI-runner noise, so the **speed** trends plot a
  machine-invariant ratio-to-fastest on a **log axis**; memory, conformance, and bundle size
  are stable, so they trend as absolute values. Hover any point for its date, value, and —
  where the run recorded one — the library version; a ringed point marks a version change.
- Every chart is a static image — inline SVG rendered at build time, no client JavaScript.

These numbers track real runs, not a hand-captured snapshot: CI publishes each run to the
orphan [`benchmark-data`](https://github.com/jbsiddall/lightning-yaml/tree/benchmark-data)
branch, and the deploy workflow overlays that history at build time — the bar charts render
the newest published document per suite, the trend charts the full history. To reproduce them
yourself, see [running the benchmarks & tests](/research/methodology/#running-the-benchmarks--tests).

## Conformance

Pass rate across the official [yaml-test-suite](https://github.com/yaml/yaml-test-suite) — the
YAML 1.2 spec operationalized as executable cases — a property of the parser, not a per-workload
measurement. Higher is better.

<ChartCard

## Bundle size

`stringify` and ships it minified — a whole-library figure, not split by operation. Measured
with `js-yaml`, `yaml` (eemeli), and `lightning-yaml`, tree-shaken and gzipped the same way a
real app would ship them. Sizes are deterministic (not machine-dependent). Lower is better.
